483 inline bool Strategy::OnExternalOrder(ExternalOrderEventMsg& msg)
484 {
485  bool result = false;
486 
487  if (msg.source() == ExternalOrderEventMsg::EXTERNAL_ORDER_SOURCE_STRATEGY_MANAGER) {
488  switch (msg.order_request_type()) {
489  case FlashProtocol::ORDER_SUBMIT_NEW: {
490  TradeActionResult taResult = trade_actions()->SendNewOrder(msg.params(), msg.account().empty() ? account_owner() : msg.account());
491  if (taResult == TRADE_ACTION_RESULT_SUCCESSFUL) {
492  result = true;
493  } else {
494  std::string message = "Client NEW Order with ID = " + msg.client_order_id() + ", Rejected with status = " + DataTypesUtils::TradeActionResultToString(taResult);
495  logger().LogToClient(Utilities::LOGLEVEL_ERROR, message);
496  }
497  break;
498  }
499  case FlashProtocol::ORDER_CANCEL: {
500  TradeActionResult taResult = trade_actions()->SendCancelOrder(msg.params().order_id);
501  if (taResult == TRADE_ACTION_RESULT_SUCCESSFUL) {
502  result = true;
503  } else {
504  std::string message = "Client CANCEL Order with ID = " + msg.client_order_id() + ", Rejected with status = " + DataTypesUtils::TradeActionResultToString(taResult);
505  logger().LogToClient(Utilities::LOGLEVEL_ERROR, message);
506  }
507  break;
508  }
509  case FlashProtocol::ORDER_REPLACE: {
510  TradeActionResult taResult = trade_actions()->SendCancelReplaceOrder(msg.params().order_id, msg.params());
511  if (taResult == TRADE_ACTION_RESULT_SUCCESSFUL) {
512  result = true;
513  } else {
514  std::string message = "Client CANCEL REPLACE Order with ID = " + msg.client_order_id() + ", Rejected with status = " + DataTypesUtils::TradeActionResultToString(taResult);
515  logger().LogToClient(Utilities::LOGLEVEL_ERROR, message);
516  }
517  break;
518  }
519  default:
520  break;
521  }
522  }
523  return result;
524 }
