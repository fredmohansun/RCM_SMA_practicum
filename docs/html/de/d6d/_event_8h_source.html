103 inline bool Event::OnEvent(EventMsg* eventMsg)
104 {
105  bool retval = false;
106  EventListenersIter end_it = m_eventListeners.end();
107  for(EventListenersIter it = m_eventListeners.begin(); it != end_it; ++it)
108  {
109  eventMsg->PushActiveEvent(this);
110 
111  SERVER_DEBUG_LOG("Sending event to listener " + it->second->event_listener_name());
112  retval |= it->second->OnEvent(eventMsg);
113  SERVER_DEBUG_LOG("Finished sending event to listener " + it->second->event_listener_name());
114 
115  eventMsg->PopActiveEvent();
116  }
117 
118  return retval;
119 }
